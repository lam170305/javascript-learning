async function getUsers() {

    try {

        loading.style.display = "block";

        const response = await fetch(
            "https://jsonplaceholder.typicode.com/posts"
        );

        const data = await response.json();

        users.innerHTML = "";

        data.forEach(post => {
            users.innerHTML += `
                <div>
                    <h2>${post.title}</h2>
                    <p>${post.body}</p>
                </div>
            `;
        });

    } catch (error) {

        console.error(error);

    } finally {

        loading.style.display = "none";

    }
}

getUsers();
